Dieu.evolveCell(int x,int y)
	bornRules = rules.getBornRules();
	surviveRules = rules.getSurviveRules();
	nbVoisinesVivante = getNbvoisinesVivanteDe(x,y);
	cellule = monde.getCell(x,y);
	if(cellule.isAlive())
		cellule.setNextTimeStatus(surviveRules[nbVoisinesVivante])
	else //morte
		cellule.setNextTimeStatus(bornRules[nbVoisinesVivante])
