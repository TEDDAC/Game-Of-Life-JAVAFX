evolution():
rules = Jeu.getRule();
celulleenvie = Jeu.getCelluleEnVie();
cellulesTraité = Jeu.getTraité()

foreach celulleenvie as cellule
	if(!traité.isIn(celulle))
		cellule.evolve(rules)
		cellulesTraité.ajouter(celulle)
		if(cellule.nextTimeStatus == 0) //juste pour l'opti
			return
		foreach(cellule.getVoisines() as voisines)
			if(!cellulesTraité.isIn(voisines))
				voisines.evolve(rules);
				cellulesTraité.ajouter(voisine);
	

evolve(rules):
	nbVoisinesVivante = getNbvoisinesVivante();
	if(isAlive())
		if(survivedRules[nbVoisinesVivantes] == TRUE) //peut survivre
			this.nextTimeStatus = 0;
		else
			this.nextTimeStatus = -1;
	else //morte
		if(bornRules[nbVoisinesVivantes] == TRUE)
			this.nextTimeStatus = 1;
		else
			this.nextTimeStatus = 0;

updateCells():
foreach cellulesTraite as cellule
	celulle.update()
	traite.remove(cellule)
cellulesTraite.clear()
